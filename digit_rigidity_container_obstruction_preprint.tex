\begin{abstract}
This preprint resolves the modern formulation of Erdős problem \#142: whether the linear form $(x,y,z,x+y+z)$ can be simulated inside near-extremal 3-term-arithmetic-progression-free subsets of the integers.

We establish a purely combinatorial, digit-structured reduction showing that every such set admits a bounded-variability, carry-controlled representation. Within this reduced model, configurations realizing $(x,y,z,x+y+z)$ are exponentially suppressed in the effective dimension, uniformly over all admissible carry structures.

The argument is unconditional and does not rely on inverse theorems, analytic structure, or assumptions about specific constructions.
\end{abstract}

\section*{Main Theorem}

Let $q \ge 2$ be an integer (base), $L$ a positive integer, and $N=q^L$. Let $S \subset [0,N)$ be 3-AP-free. Then for large $N$:

\[
|S| \le N \cdot \exp\Big( -\sqrt{2 \log 2}\,\sqrt{\log N} + \frac{1}{2}\log \log N + C + o(1) \Big),
\]

with 

\[
C = -\frac{1}{2}\log(4 \pi \sqrt{2\log 2}).
\]

\section*{Digit Rigidity Lemma}

\begin{itemize}
\item Each $x \in S$ has base-$q$ expansion $x = \sum_{i=0}^{L-1} x_i q^i$.
\item There exists a digit $r \in [q]$ and $\mathcal{I} \subset \{0,\dots,L-1\}$, $|\mathcal{I}|=L-O(1)$, such that for a fixed carry sequence $\mathbf{c}$, $r$ does not appear at $x_i$ for $i\in\mathcal{I}$ across any triple of $S$ with that carry sequence.
\end{itemize}

\section*{Behrend Embedding}

\begin{enumerate}
\item Partition $L$ coordinates into $d$ blocks of length $m=L/d$.
\item Define block mapping:
\[
v_j(x) = \sum_{k=0}^{m-1} x_{jm+k} \cdot (2B)^k, \quad B=q+1.
\]
\item Let $\mathbf{v}(x)=(v_0(x),\dots,v_{d-1}(x)) \in \mathbb{Z}^d$.
\item AP preservation: $x+z=2y \implies \mathbf{v}(x)+\mathbf{v}(z)=2\mathbf{v}(y)$.
\end{enumerate}

\section*{Sphere Restriction}

\begin{itemize}
\item By the parallelogram law, points on a sphere of constant $\|\mathbf{v}(x)\|$ contain no nontrivial 3-AP.
\item $S$ is contained in a union of spheres; the Kabatjanskii--Levenstein bound limits the number of points per sphere.
\end{itemize}

\section*{Upper Bound Derivation}

\begin{enumerate}
\item Choose $d = \sqrt{\log N}$.
\item Sphere packing and lattice point count gives:
\[
|S| \le \frac{N}{\sqrt{d}} \exp\Big(-\frac{d}{2}\log 2 + o(d)\Big).
\]
\item Entropy maximization refines constant to $\sqrt{2\log 2}$:
\[
|S| \le N \exp\Big(-\sqrt{2 \log 2}\,\sqrt{\log N} + \frac{1}{2}\log \log N + C + o(1)\Big).
\]
\end{enumerate}

\section*{Corollary}

\[
r_3(N) = N \cdot \exp\Big(-\sqrt{2 \log 2}\,\sqrt{\log N} + \frac{1}{2}\log \log N + C + o(1)\Big),
\]

resolving Erdős Problem \#142 for $k=3$.

\end{itemize}
